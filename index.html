How keys work with your app User enters key + Gmail in your WinForms login form → app calls Firebase Cloud Function → server checks key validity, expiry, and device HWID → returns status. 1 key = 1 device. Key locks to HWID on first use. Expired keys return "expired" status and app closes.
GenxLicense.cs — paste into your WinForms project (requires Newtonsoft.Json NuGet) 
// GenxLicense.cs — C# WinForms License Validator
// NuGet: Install-Package Newtonsoft.Json
// NuGet: Install-Package System.Management (for HWID)

using System;
using System.Net.Http;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using Newtonsoft.Json;

namespace GenxApp
{
    public static class GenxLicense
    {
        // ▶ Replace with your Firebase Cloud Function URL
        private const string API =
            "https://YOUR_REGION-YOUR_PROJECT.cloudfunctions.net/validateKey";

        // Validate key against server
        public static async Task<LicenseResult> Validate(string key, string email)
        {
            string hwid = GetHWID();
            using var client = new HttpClient { Timeout = TimeSpan.FromSeconds(12) };
            var body = JsonConvert.SerializeObject(new { key, email, hwid });
            var content = new StringContent(body, Encoding.UTF8, "application/json");
            try {
                var resp = await client.PostAsync(API, content);
                var json = await resp.Content.ReadAsStringAsync();
                return JsonConvert.DeserializeObject<LicenseResult>(json)
                       ?? new LicenseResult { status = "error" };
            } catch (Exception ex) {
                return new LicenseResult { status = "offline", message = ex.Message };
            }
        }

        // SHA-256(MotherboardSerial | CPUId) → 24-char uppercase HWID
        public static string GetHWID()
        {
            string mb  = GetWMI("Win32_BaseBoard",   "SerialNumber");
            string cpu = GetWMI("Win32_Processor",   "ProcessorId");
            string raw = mb + "|" + cpu;
            using var sha = System.Security.Cryptography.SHA256.Create();
            var bytes = sha.ComputeHash(Encoding.UTF8.GetBytes(raw));
            return BitConverter.ToString(bytes).Replace("-", "").ToUpper().Substring(0, 24);
        }

        private static string GetWMI(string cls, string prop)
        {
            try {
                var searcher = new System.Management.ManagementObjectSearcher(
                    "SELECT * FROM " + cls);
                foreach (System.Management.ManagementObject o in searcher.Get())
                    return o[prop]?.ToString()?.Trim() ?? "UNKNOWN";
            } catch {}
            return "UNKNOWN";
        }
    }

    public class LicenseResult
    {
        public string status    { get; set; } // active|expired|invalid|device_mismatch|offline|error
        public int    daysLeft  { get; set; }
        public string plan      { get; set; }
        public string expiresAt { get; set; }
        public string message   { get; set; }
    }

    // ── LOGIN FORM ──
    public partial class LoginForm : Form
    {
        public LoginForm()
        {
            InitializeComponent();
            // Set form properties:
            this.Text        = "Krypto Studio — Login";
            this.Size        = new System.Drawing.Size(420, 340);
            this.StartPosition = FormStartPosition.CenterScreen;
            this.FormBorderStyle = FormBorderStyle.FixedDialog;
            this.MaximizeBox = false;
            BuildUI();
        }

        private TextBox txtEmail, txtKey;
        private Button  btnLogin;
        private Label   lblStatus;

        private void BuildUI()
        {
            var lbl1 = new Label { Text="Gmail", Location=new System.Drawing.Point(30,40),
                AutoSize=true, ForeColor=System.Drawing.Color.Gray };
            txtEmail = new TextBox { Location=new System.Drawing.Point(30,58),
                Width=340, Font=new System.Drawing.Font("Consolas",11) };

            var lbl2 = new Label { Text="License Key", Location=new System.Drawing.Point(30,100),
                AutoSize=true, ForeColor=System.Drawing.Color.Gray };
            txtKey = new TextBox { Location=new System.Drawing.Point(30,118),
                Width=340, Font=new System.Drawing.Font("Consolas",11),
                CharacterCasing=CharacterCasing.Upper };

            btnLogin = new Button { Text="Activate", Location=new System.Drawing.Point(30,164),
                Width=340, Height=38,
                Font=new System.Drawing.Font("Syne",11,System.Drawing.FontStyle.Bold),
                BackColor=System.Drawing.Color.FromArgb(191,127,255),
                ForeColor=System.Drawing.Color.White, FlatStyle=FlatStyle.Flat };
            btnLogin.FlatAppearance.BorderSize=0;
            btnLogin.Click += BtnLogin_Click;

            lblStatus = new Label { Location=new System.Drawing.Point(30,215),
                Width=340, AutoSize=false, Height=60, Font=new System.Drawing.Font("Consolas",10),
                ForeColor=System.Drawing.Color.Gray, TextAlign=System.Drawing.ContentAlignment.MiddleCenter };

            var link = new LinkLabel { Text="Get a key at https://roii-x.github.io/genx/",
                Location=new System.Drawing.Point(30,280), AutoSize=true,
                Font=new System.Drawing.Font("Consolas",9) };
            link.Click += (s,e) => System.Diagnostics.Process.Start("https://roii-x.github.io/genx/");

            this.Controls.AddRange(new Control[]{lbl1,txtEmail,lbl2,txtKey,btnLogin,lblStatus,link});
        }

        private async void BtnLogin_Click(object sender, EventArgs e)
        {
            string email = txtEmail.Text.Trim().ToLower();
            string key   = txtKey.Text.Trim().ToUpper();

            if (string.IsNullOrEmpty(email) || !email.Contains("@")) {
                lblStatus.ForeColor = System.Drawing.Color.Tomato;
                lblStatus.Text = "Enter a valid Gmail address."; return;
            }
            if (key.Length < 19) {
                lblStatus.ForeColor = System.Drawing.Color.Tomato;
                lblStatus.Text = "Enter a valid license key."; return;
            }

            btnLogin.Enabled = false;
            btnLogin.Text    = "Checking...";
            lblStatus.ForeColor = System.Drawing.Color.Gray;
            lblStatus.Text   = "Contacting server...";

            var result = await GenxLicense.Validate(key, email);

            btnLogin.Enabled = true;
            btnLogin.Text    = "Activate";

            switch (result.status)
            {
                case "active":
                    lblStatus.ForeColor = System.Drawing.Color.LimeGreen;
                    lblStatus.Text = $"✅ Activated! Plan: {result.plan} | Days left: {result.daysLeft}";
                    await Task.Delay(1200);
                    this.Hide();
                    new MainForm().Show();  // ← Open your main app form here
                    break;
                case "expired":
                    lblStatus.ForeColor = System.Drawing.Color.Orange;
                    lblStatus.Text = $"❌ License expired on {result.expiresAt}
Renew at roii-x.github.io/genx/";
                    break;
                case "invalid":
                    lblStatus.ForeColor = System.Drawing.Color.Tomato;
                    lblStatus.Text = "⛔ Invalid license key. Check for typos.";
                    break;
                case "device_mismatch":
                    lblStatus.ForeColor = System.Drawing.Color.Tomato;
                    lblStatus.Text = "⛔ Key is locked to a different device.";
                    break;
                case "email_mismatch":
                    lblStatus.ForeColor = System.Drawing.Color.Tomato;
                    lblStatus.Text = "⛔ Gmail does not match this key.";
                    break;
                case "disabled":
                    lblStatus.ForeColor = System.Drawing.Color.Gray;
                    lblStatus.Text = "⛔ This key has been disabled by admin.";
                    break;
                case "offline":
                    lblStatus.ForeColor = System.Drawing.Color.Orange;
                    lblStatus.Text = "⚠️ No internet connection. Cannot verify.";
                    break;
                default:
                    lblStatus.ForeColor = System.Drawing.Color.Gray;
                    lblStatus.Text = $"Error: {result.message ?? "Unknown error"}";
                    break;
            }
        }
    }
}
